## 2. JWT Payload (Claims)

Dùng **standard claims** của PyJWT + custom claims:

```python
# auth/utils.py
from datetime import datetime, timedelta, timezone
import uuid
import jwt
from app.config import settings


def create_access_token(
    user_id: int,
    email: str,
    is_superuser: bool,
    group_ids: list[int],
) -> str:
    """Access token: ngắn hạn (30 phút), chứa claims đầy đủ.
    Nhận từng field thay vì User object — tách dependency, dễ test.
    """
    now = datetime.now(timezone.utc)
    payload = {
        # --- Standard Claims (RFC 7519) ---
        "sub": str(user_id),          # Subject: user_id (phải là string theo chuẩn JWT)
        "iat": now,                   # Issued At: PyJWT tự convert sang Unix timestamp
        "exp": now + timedelta(minutes=settings.ACCESS_TOKEN_EXPIRE_MINUTES),
        "jti": uuid.uuid4().hex,      # JWT ID: unique serial number mỗi token (chống replay)
        "type": "access",             # Phân biệt access vs refresh vs verify-email
        # --- Custom Claims ---
        "email": email,
        "is_superuser": is_superuser,
        "group_ids": group_ids,       # Tránh query DB mỗi request (xem Section 7)
    }
    return jwt.encode(payload, settings.SECRET_KEY, algorithm=settings.JWT_ALGORITHM)


def create_refresh_token(user_id: int) -> tuple[str, str, datetime]:
    """Refresh token: dài hạn (30 ngày), chỉ chứa sub + jti.
    Returns: (jwt_token, jti, expires_at) — jti lưu vào DB để revoke khi logout.
    """
    now = datetime.now(timezone.utc)
    jti = uuid.uuid4().hex
    expires_at = now + timedelta(days=settings.REFRESH_TOKEN_EXPIRE_DAYS)
    payload = {
        "sub": str(user_id),
        "exp": expires_at,
        "iat": now,
        "jti": jti,
        "type": "refresh",
    }
    token = jwt.encode(payload, settings.SECRET_KEY, algorithm=settings.JWT_ALGORITHM)
    return token, jti, expires_at


def create_verify_email_token(user_id: int) -> str:
    """JWT cho email verification (24h). Dùng type='verify-email'."""
    now = datetime.now(timezone.utc)
    payload = {
        "sub": str(user_id),
        "type": "verify-email",
        "iat": now,
        "exp": now + timedelta(hours=24),
    }
    return jwt.encode(payload, settings.SECRET_KEY, algorithm=settings.JWT_ALGORITHM)


def decode_token(token: str) -> dict:
    """Decode và validate JWT. Dùng chung cho access, refresh, verify-email.
    Raises:
        jwt.ExpiredSignatureError   → token hết hạn  → 401
        jwt.InvalidTokenError       → token sai      → 401
    """
    return jwt.decode(
        token,
        settings.SECRET_KEY,
        algorithms=[settings.JWT_ALGORITHM],
        options={"require": ["sub", "exp", "iat", "type"]},  # require "type" (không phải "jti" — verify-email không có jti)
    )
```

> **Tại sao `sub` là `str`?**  
> RFC 7519 định nghĩa `sub` là `StringOrURI`. PyJWT decode trả về đúng kiểu gốc, nên parse thủ công: `user_id = int(payload["sub"])`.

```python
# auth/models.py — CMS Core models
from sqlalchemy import Column, Integer, String, Boolean, DateTime, Text, ForeignKey, Table, func
from sqlalchemy.orm import relationship
from sqlalchemy import UniqueConstraint
from app.database import Base

# --- Bảng trung gian M2M ---
user_groups = Table(
    "user_groups", Base.metadata,
    Column("user_id", Integer, ForeignKey("users.id"), primary_key=True),
    Column("group_id", Integer, ForeignKey("groups.id"), primary_key=True),
)

class User(Base):
    __tablename__ = "users"

    id                   = Column(Integer, primary_key=True, index=True)
    email                = Column(String(255), unique=True, nullable=False, index=True)
    password_hash        = Column(String(255), nullable=False)        # KHÔNG expose qua API
    full_name            = Column(String(255), nullable=True)
    is_active            = Column(Boolean, default=True, nullable=False)
    is_superuser         = Column(Boolean, default=False, nullable=False)
    email_verified       = Column(Boolean, default=False, nullable=False)   # xác minh email
    password_reset_token = Column(String(255), nullable=True)               # token reset mật khẩu (UUID, có TTL)
    password_reset_at    = Column(DateTime(timezone=True), nullable=True)   # thời điểm tạo reset token
    created_at           = Column(DateTime(timezone=True), server_default=func.now())
    updated_at           = Column(DateTime(timezone=True), onupdate=func.now())

    groups = relationship("Group", secondary=user_groups, lazy="selectin")

class Group(Base):
    __tablename__ = "groups"

    id          = Column(Integer, primary_key=True, index=True)
    name        = Column(String(255), unique=True, nullable=False)
    description = Column(Text, nullable=True)
    created_at  = Column(DateTime(timezone=True), server_default=func.now())

class GroupPermission(Base):
    __tablename__  = "group_permissions"
    __table_args__ = (UniqueConstraint("group_id", "module_name", "action"),)

    id          = Column(Integer, primary_key=True, index=True)
    group_id    = Column(Integer, ForeignKey("groups.id"), nullable=False, index=True)
    module_name = Column(String(255), nullable=False)   # "products", "orders"...
    action      = Column(String(50), nullable=False)     # list|read|create|update|archive|restore|delete|bulk
    allowed     = Column(Boolean, default=True)
    created_at  = Column(DateTime(timezone=True), server_default=func.now())

    group = relationship("Group")


class RefreshToken(Base):
    """Lưu refresh token trong DB để có thể revoke khi logout."""
    __tablename__ = "refresh_tokens"

    id         = Column(Integer, primary_key=True, index=True)
    user_id    = Column(Integer, ForeignKey("users.id"), nullable=False, index=True)
    jti        = Column(String(255), unique=True, nullable=False, index=True)  # JWT ID của refresh token
    expires_at = Column(DateTime(timezone=True), nullable=False)
    revoked    = Column(Boolean, default=False, nullable=False)                # True khi logout
    created_at = Column(DateTime(timezone=True), server_default=func.now())

    user = relationship("User")
```

> `lazy="selectin"` → khi load User, tự load groups trong 1 query bổ sung (không N+1).  
> **Không expose `password_hash` qua API** — đây là internal field, đổi qua `/auth/set-password`.  
> CMS Core endpoints (CRUD `/users/`, `/groups/`, `/group-permissions/`) viết tay trong `system/router.py` — **superuser-only**.

## 4. Password & Token Hashing (`auth/utils.py`)

```python
import bcrypt
import hashlib

def hash_password(plain: str) -> str:
    return bcrypt.hashpw(plain.encode(), bcrypt.gensalt()).decode()

def verify_password(plain: str, hashed: str) -> bool:
    return bcrypt.checkpw(plain.encode(), hashed.encode())

def hash_token(token: str) -> str:
    """Hash reset token bằng SHA-256 trước khi lưu DB.
    Không dùng bcrypt vì token đã là UUID (high entropy) — không cần cost factor.
    """
    return hashlib.sha256(token.encode()).hexdigest()

def verify_token(plain_token: str, hashed_token: str) -> bool:
    """So sánh token plaintext với hash trong DB."""
    return hash_token(plain_token) == hashed_token
```

> **Tại sao `bcrypt` cho password?** bcrypt có cost factor (rounds) — chống brute-force tốt hơn MD5/SHA.  
> **Tại sao `SHA-256` cho reset token?** Token là UUID (128-bit entropy) — không cần slow hash. Chỉ cần DB lộ thì attacker không dùng được.  
> Cài: `pip install bcrypt`

## 6. Auth Endpoints (`auth/router.py`)

> Dùng `Annotated` dependencies thay vì `getattr(request.state)`:
> - `DBSession = Annotated[AsyncSession, Depends(get_db)]`
> - `CurrentUserId = Annotated[int, Depends(require_auth)]` — tự 401 nếu chưa đăng nhập
> - `SuperUserId = Annotated[int, Depends(require_superuser)]` — tự 403 nếu không phải superuser

```python
# auth/router.py
from auth.dependencies import CurrentUserId
from app.database import DBSession

router = APIRouter(prefix="/auth", tags=["auth"])

def _success(data=None, message=None) -> dict:
    return {"success": True, "data": data, "message": message}


# POST /auth/login — đăng nhập, trả dual tokens
@router.post("/login")
async def login(body: LoginRequest, db: DBSession) -> dict:
    result = await db.execute(select(User).where(User.email == body.email))
    user = result.scalar_one_or_none()

    if not user or not verify_password(body.password, user.password_hash):
        raise HTTPException(status_code=401, detail="Email hoặc mật khẩu không đúng")

    if not user.is_active:
        raise HTTPException(status_code=403, detail="Tài khoản đã bị vô hiệu hoá")

    group_ids = [g.id for g in user.groups]

    access_token = create_access_token(
        user_id=user.id, email=user.email,
        is_superuser=user.is_superuser, group_ids=group_ids,
    )
    refresh_token_str, jti, expires_at = create_refresh_token(user.id)

    # Lưu refresh token vào DB
    db.add(RefreshToken(user_id=user.id, jti=jti, expires_at=expires_at))
    await db.commit()

    login_data = LoginResponse(
        access_token=access_token, refresh_token=refresh_token_str,
        expires_in=settings.ACCESS_TOKEN_EXPIRE_MINUTES * 60,
        user=UserInfo.model_validate(user),
    )
    return _success(data=login_data.model_dump())


# POST /auth/logout — revoke 1 refresh token (single-device)
@router.post("/logout")
async def logout(body: RefreshTokenRequest, db: DBSession, _user_id: CurrentUserId) -> dict:
    """Nhận refresh_token trong body → revoke chỉ token đó (single-device logout)."""
    try:
        payload = decode_token(body.refresh_token)
        jti = payload.get("jti")
    except Exception:
        return _success(message="Đã đăng xuất")

    if jti:
        result = await db.execute(select(RefreshToken).where(RefreshToken.jti == jti))
        db_token = result.scalar_one_or_none()
        if db_token:
            db_token.revoked = True
            await db.commit()

    return _success(message="Đã đăng xuất")
    # Frontend: xoá access_token + refresh_token khỏi localStorage/cookie


# POST /auth/refresh — đổi refresh_token lấy access_token mới
@router.post("/refresh")
async def refresh(body: RefreshTokenRequest, db: DBSession) -> dict:
    try:
        payload = decode_token(body.refresh_token)
    except Exception:
        raise HTTPException(status_code=401, detail="Refresh token không hợp lệ hoặc đã hết hạn")

    if payload.get("type") != "refresh":
        raise HTTPException(status_code=401, detail="Refresh token không hợp lệ hoặc đã hết hạn")

    jti = payload.get("jti")
    result = await db.execute(select(RefreshToken).where(RefreshToken.jti == jti))
    db_token = result.scalar_one_or_none()

    if not db_token or db_token.revoked:
        raise HTTPException(status_code=401, detail="Refresh token không hợp lệ hoặc đã bị thu hồi")

    user_id = int(payload["sub"])
    result = await db.execute(select(User).where(User.id == user_id))
    user = result.scalar_one_or_none()

    if not user or not user.is_active:
        raise HTTPException(status_code=401, detail="Tài khoản không tồn tại hoặc đã bị vô hiệu hoá")

    group_ids = [g.id for g in user.groups]
    access_token = create_access_token(
        user_id=user.id, email=user.email,
        is_superuser=user.is_superuser, group_ids=group_ids,
    )

    token_data = TokenResponse(access_token=access_token, expires_in=settings.ACCESS_TOKEN_EXPIRE_MINUTES * 60)
    return _success(data=token_data.model_dump())


# GET /auth/me — thông tin user hiện tại
@router.get("/me")
async def me(request: Request, db: DBSession, _user_id: CurrentUserId) -> dict:
    result = await db.execute(select(User).where(User.id == request.state.user_id))
    user = result.scalar_one_or_none()
    if not user:
        raise HTTPException(status_code=401, detail="User không tồn tại")
    return _success(data=UserInfo.model_validate(user).model_dump())


# POST /auth/set-password — đổi mật khẩu (biết password cũ)
@router.post("/set-password")
async def set_password(body: SetPasswordRequest, db: DBSession, _user_id: CurrentUserId, request: Request) -> dict:
    result = await db.execute(select(User).where(User.id == request.state.user_id))
    user = result.scalar_one_or_none()
    if not user:
        raise HTTPException(status_code=401)
    if not verify_password(body.current_password, user.password_hash):
        raise HTTPException(status_code=400, detail="Mật khẩu hiện tại không đúng")
    user.password_hash = hash_password(body.new_password)
    await db.commit()
    return _success(message="Đổi mật khẩu thành công")


# POST /auth/register — đăng ký tài khoản mới
@router.post("/register", status_code=201)
async def register(body: RegisterRequest, db: DBSession) -> dict:
    if not settings.ALLOW_PUBLIC_REGISTER:
        raise HTTPException(status_code=403, detail="Đăng ký bị vô hiệu")

    result = await db.execute(select(User).where(User.email == body.email))
    if result.scalar_one_or_none():
        raise HTTPException(status_code=409, detail="Email đã được sử dụng")

    user = User(
        email=body.email,
        password_hash=hash_password(body.password),
        full_name=body.full_name,
        email_verified=False,
    )
    db.add(user)
    await db.commit()
    await db.refresh(user)

    # Gửi email xác minh (sync — Phase 0)
    verify_token = create_verify_email_token(user.id)
    send_verify_email(user.email, verify_token)    # sync call (Phase 0)

    return _success(
        data={"id": user.id, "email": user.email},
        message="Đăng ký thành công. Kiểm tra email để xác minh.",

# POST /auth/verify-email — xác minh email qua token
@router.post("/verify-email")
async def verify_email(body: VerifyEmailRequest, db: DBSession) -> dict:
    try:
        payload = decode_token(body.token)
    except Exception:
        raise HTTPException(status_code=400, detail="Token không hợp lệ hoặc đã hết hạn")

    if payload.get("type") != "verify-email":
        raise HTTPException(status_code=400, detail="Token không hợp lệ hoặc đã hết hạn")

    user_id = int(payload["sub"])
    result = await db.execute(select(User).where(User.id == user_id))
    user = result.scalar_one_or_none()
    if not user:
        raise HTTPException(status_code=400, detail="Token không hợp lệ hoặc đã hết hạn")

    user.email_verified = True
    await db.commit()
    return _success(message="Email đã được xác minh")


# POST /auth/forgot-password — gửi email reset mật khẩu
@router.post("/forgot-password")
async def forgot_password(body: ForgotPasswordRequest, db: DBSession) -> dict:
    result = await db.execute(select(User).where(User.email == body.email))
    user = result.scalar_one_or_none()

    if user:
        plaintext_token, token_hash = generate_reset_token()
        user.password_reset_token = token_hash
        user.password_reset_at = datetime.now(timezone.utc)
        await db.commit()
        send_reset_email(user.email, plaintext_token)    # sync call (Phase 0)

    # Luôn 200 — không lộ email nào tồn tại
    return _success(message="Nếu email tồn tại, link reset đã được gửi")


# POST /auth/reset-password — đặt lại mật khẩu bằng reset token
@router.post("/reset-password")
async def reset_password(body: ResetPasswordRequest, db: DBSession) -> dict:
    token_hash = hash_token(body.token)
    result = await db.execute(select(User).where(User.password_reset_token == token_hash))
    user = result.scalar_one_or_none()

    if not user:
        raise HTTPException(status_code=400, detail="Token không hợp lệ hoặc đã hết hạn")

    # Check TTL: 1 giờ
    if user.password_reset_at:
        elapsed = datetime.now(timezone.utc) - user.password_reset_at.replace(tzinfo=timezone.utc)
        if elapsed > timedelta(hours=1):
            user.password_reset_token = None
            user.password_reset_at = None
            await db.commit()
            raise HTTPException(status_code=400, detail="Token không hợp lệ hoặc đã hết hạn")

    user.password_hash = hash_password(body.new_password)
    user.password_reset_token = None
    user.password_reset_at = None
    await db.commit()

    return _success(message="Đặt lại mật khẩu thành công")
```

## 7. JWTMiddleware (`auth/middleware.py`)

```python
# auth/middleware.py
import jwt
from starlette.middleware.base import BaseHTTPMiddleware
from starlette.requests import Request
from starlette.responses import JSONResponse
from auth.utils import decode_token

# Paths không cần token (middleware cho đi qua, TRCFBaseModule không check)
PUBLIC_PATHS = {
    "/",
    "/docs",
    "/openapi.json",
    "/redoc",
    "/auth/login",
    "/auth/register",
    "/auth/verify-email",
    "/auth/forgot-password",
    "/auth/reset-password",
    "/auth/refresh",
}

class JWTMiddleware(BaseHTTPMiddleware):
    async def dispatch(self, request: Request, call_next: RequestResponseEndpoint) -> Response:
        # Default: no auth (null-safe cho TRCFBaseModule)
        request.state.user_id = None
        request.state.user_groups = []
        request.state.user_is_superuser = False

        # Skip auth for public paths
        path = request.url.path.rstrip("/")
        if path in PUBLIC_PATHS or path.startswith("/docs") or path.startswith("/redoc"):
            return await call_next(request)

        # Extract token from Authorization header
        auth_header = request.headers.get("Authorization")
        if auth_header and auth_header.startswith("Bearer "):
            token = auth_header[7:]
            try:
                payload = decode_token(token)

                # CRITICAL: reject refresh tokens used as access
                if payload.get("type") != "access":
                    return await call_next(request)  # inject null → endpoint tự 401

                request.state.user_id = int(payload["sub"])
                request.state.user_groups = payload.get("group_ids", [])
                request.state.user_is_superuser = payload.get("is_superuser", False)

            except Exception:
                # Invalid/expired token → proceed without auth (endpoints will check)
                pass

        return await call_next(request)
```

> **Thiết kế:** Middleware inject `null state` trước → TRCFBaseModule/`CurrentUserId` dependency quyết định có 401 hay không.
> Token sai hoặc hết hạn → middleware inject null state, endpoint tự xử lý 401.
> Token type không phải `access` → inject null state (không cho refresh token dùng làm Bearer).

## 8. Permission Check (`base/permissions.py`)

```python
# base/permissions.py — dùng trong router_factory.py cho mỗi CRUD endpoint
async def check_permission(
    request: Request,
    db: AsyncSession,
    module_name: str,
    action: str,
    *,
    require_auth: bool = True,
    public_actions: list[str] | None = None,
) -> int | None:
    """
    Returns: user_id (int) nếu authenticated, None nếu public access.
    Raises: 401 nếu chưa đăng nhập, 403 nếu không có quyền.
    """
    public_actions = public_actions or []

    user_id = getattr(request.state, "user_id", None)
    is_superuser = getattr(request.state, "user_is_superuser", False)
    group_ids = getattr(request.state, "user_groups", [])

    # 1. Superuser bypass — không cần check gì thêm
    if is_superuser and user_id is not None:
        return user_id

    # 2. Public action → cho qua (ví dụ: _public_actions = ["list", "read"])
    if action in public_actions:
        return user_id  # None nếu chưa đăng nhập

    # 3. Require auth nhưng chưa đăng nhập → 401
    if require_auth and user_id is None:
        raise HTTPException(status_code=401, detail="Chưa đăng nhập")

    # 4. Module không require auth và chưa đăng nhập
    if not require_auth and user_id is None:
        return None

    # 5. Đã đăng nhập → check group permissions
    if not group_ids:
        raise HTTPException(status_code=403, detail=f"Không có quyền {action} trên {module_name}")

    result = await db.execute(
        select(GroupPermission).where(
            GroupPermission.group_id.in_(group_ids),
            GroupPermission.module_name == module_name,
            GroupPermission.action == action,
            GroupPermission.allowed == True,
        )
    )
    if not result.scalar_one_or_none():
        raise HTTPException(status_code=403, detail=f"Không có quyền {action} trên {module_name}")

    return user_id
```

```python
# auth/email.py
import logging
import smtplib
from email.mime.multipart import MIMEMultipart
from email.mime.text import MIMEText
from app.config import settings

logger = logging.getLogger(__name__)


def _send_email(to_email: str, subject: str, html_body: str) -> None:
    """Sync SMTP sender. Falls back to console log if SMTP_HOST is empty."""
    if not settings.SMTP_HOST:
        logger.info("📧 EMAIL (dev mode) To: %s  Subject: %s", to_email, subject)
        return

    msg = MIMEMultipart("alternative")
    msg["Subject"] = subject
    msg["From"] = settings.SMTP_FROM
    msg["To"] = to_email
    msg.attach(MIMEText(html_body, "html"))

    try:
        server = smtplib.SMTP(settings.SMTP_HOST, settings.SMTP_PORT)
        if settings.SMTP_TLS:
            server.starttls()
        if settings.SMTP_USER:
            server.login(settings.SMTP_USER, settings.SMTP_PASSWORD)
        server.sendmail(settings.SMTP_FROM, to_email, msg.as_string())
        server.quit()
    except Exception:
        logger.exception("Failed to send email to %s", to_email)
        raise


def send_reset_email(to_email: str, reset_token: str) -> None:
    """Gửi email reset mật khẩu — sync."""
    reset_url = f"{settings.FRONTEND_URL}/reset-password?token={reset_token}"
    html = f"<h2>Reset mật khẩu — CoffeeTree</h2><p><a href='{reset_url}'>{reset_url}</a></p>"
    _send_email(to_email, "Reset mật khẩu — CoffeeTree", html)

## 13. Quyết Định Thiết Kế

| Vấn đề | Quyết định | Lý do |
|---|---|---|
| Algorithm | **HS256** | Đơn giản, không cần key pair, đủ cho internal API |
| `group_ids` trong token? | **Có** | Tránh query DB mỗi request. Trade-off: nhóm thay đổi phải đợi access_token hết hạn (30 phút) hoặc refresh |
| `jti` trong token? | **Có** | UUID duy nhất mỗi token. Refresh token dùng `jti` để revoke trong DB |
| Replay attack protection? | **Chưa (Phase 0)** | Cần lưu `jti` vào Redis/DB để check — để lại Phase 2+ |
| Refresh token? | **Có (Phase 0)** | access_token 30 phút + refresh_token 30 ngày. Frontend tự refresh, user không cần đăng nhập lại |
| Token blacklist (logout)? | **Revoke DB (Phase 0)** | Logout set `revoked=True` cho refresh token trong DB. Access token tự hết hạn sau 30 phút |
| `type` claim? | **Có** | `"access"` hoặc `"refresh"` — phân biệt 2 loại token, tránh dùng sai |
| `is_superuser` trong token? | **Có** | Middleware bypass nhanh không query DB |
| Password hashing | **bcrypt** | Cost factor chống brute-force, industry standard |
| Middleware inject null state? | **Có** | TRCFBaseModule tự quyết định 401 dựa `_require_auth` — tách bạch logic |
| External SSO (Auth0/Google)? | **Không (Phase 0)** | Dùng `PyJWKClient` khi cần — fetch public key từ JWKS endpoint tự động |
| Đăng ký (Register)? | **Có (Phase 0)** | Kiểm soát bởi `ALLOW_PUBLIC_REGISTER` flag (default: `false` cho F&B POS) |
| Forgot/Reset password? | **Có (Phase 0)** | UUID token gửi user, SHA-256 hash lưu DB + TTL 1 giờ |
| Email verification? | **Có (Phase 0)** | JWT ngắn hạn (24h) cho link xác minh. Dev mode: log console |
| SMTP email? | **SMTP cơ bản (Phase 0)** | Dev mode: print console. Production: cấu hình trong `.env` |

            db.add(SystemSetting(
                module_name=module_cls._module_name,
                key=s["key"], label=s.get("label"), value=str(s.get("default", "")),
                value_type=s.get("type", "string"),
            ))
```

## 21. Session Management (Phase 1)

> User xem và quản lý refresh tokens (sessions/devices) đang hoạt động.

### Endpoints

| # | Method | Endpoint | Auth | Mô tả |
|---|--------|----------|------|--------|
| 1 | `GET` | `/auth/sessions` | Required | List refresh tokens đang active (chưa revoke, chưa hết hạn) |
| 2 | `POST` | `/auth/sessions/{jti}/revoke` | Required | Revoke 1 session cụ thể |
| 3 | `POST` | `/auth/logout-all` | Required | Revoke tất cả sessions (đăng xuất mọi thiết bị) |

```python
# GET /auth/sessions — list active sessions
@router.get("/sessions")
async def list_sessions(db: DBSession, _user_id: CurrentUserId, request: Request) -> dict:
    result = await db.execute(
        select(RefreshToken).where(
            RefreshToken.user_id == request.state.user_id,
            RefreshToken.revoked == False,
            RefreshToken.expires_at > datetime.now(timezone.utc),
        ).order_by(RefreshToken.created_at.desc())
    )
    tokens = result.scalars().all()
    return _success(data=[
        {"jti": t.jti, "created_at": t.created_at.isoformat(), "expires_at": t.expires_at.isoformat()}
        for t in tokens
    ])


# POST /auth/sessions/{jti}/revoke — revoke 1 session
@router.post("/sessions/{jti}/revoke")
async def revoke_session(jti: str, db: DBSession, _user_id: CurrentUserId, request: Request) -> dict:
    result = await db.execute(
        select(RefreshToken).where(
            RefreshToken.jti == jti,
            RefreshToken.user_id == request.state.user_id,
        )
    )
    token = result.scalar_one_or_none()
    if not token:
        raise HTTPException(status_code=404, detail="Session không tồn tại")
    token.revoked = True
    await db.commit()
    return _success(message="Session đã bị thu hồi")


# POST /auth/logout-all — đăng xuất mọi thiết bị
@router.post("/logout-all")
async def logout_all(db: DBSession, _user_id: CurrentUserId, request: Request) -> dict:
    from sqlalchemy import update
    await db.execute(
        update(RefreshToken)
        .where(RefreshToken.user_id == request.state.user_id, RefreshToken.revoked == False)
        .values(revoked=True)
    )
    await db.commit()
    return _success(message="Đã đăng xuất mọi thiết bị")
```

> **Lưu ý:** Sau khi implement, cập nhật `PUBLIC_PATHS` trong middleware và bảng endpoints ở Section 16.

## 22. Permission Wildcards (Phase 1)

> Hiện tại permissions theo `(module_name, action)` cụ thể.
> Cần hỗ trợ wildcard để đơn giản hoá quản lý quyền cho nhiều modules.

### Wildcard Rules

| Pattern | Ý nghĩa |
|---------|---------|
| `("*", "read")` | Đọc tất cả modules |
| `("products", "*")` | Full quyền trên products |
| `("*", "*")` | Full quyền tất cả (tương đương superuser but trackable) |

### Cập nhật `check_permission()`

```python
# base/permissions.py — thêm wildcard matching
async def check_permission(...):
    ...
    # Check group permissions (với wildcard support)
    result = await db.execute(
        select(GroupPermission).where(
            GroupPermission.group_id.in_(group_ids),
            GroupPermission.module_name.in_([module_name, "*"]),  # wildcard module
            GroupPermission.action.in_([action, "*"]),            # wildcard action
            GroupPermission.allowed == True,
        )
    )
    if not result.scalar_one_or_none():
        raise HTTPException(status_code=403, detail=f"Không có quyền {action} trên {module_name}")
```

### Ưu tiên permission

Khi có nhiều rules match (vd: `("*", "read")` + `("products", "read", allowed=False)`):

1. **Specific > Wildcard:** `(module, action)` > `(module, *)` > `(*, action)` > `(*, *)`
2. **Deny wins:** Nếu có 1 rule `allowed=False` cùng mức specific → deny

```python
# Permission resolution order:
# 1. (module_name, action)     — most specific
# 2. (module_name, "*")        — module wildcard
# 3. ("*", action)             — action wildcard
# 4. ("*", "*")                — global wildcard
# First match wins. If explicit deny exists at same level, deny wins.
```
